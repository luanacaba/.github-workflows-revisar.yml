name: Revisar LADE

on:
  schedule:
    - cron: "*/20 * * * *"

  workflow_dispatch:

jobs:

  revisar:
    runs-on: ubuntu-latest

    steps:

    - uses: actions/checkout@v4

    - uses: actions/setup-python@v5
      with:
        python-version: "3.12"

    - run: |
        pip install -r requirements.txt
        playwright install chromium

    - run: python bot.py
      env:
        TELEGRAM_TOKEN: ${{ secrets.TELEGRAM_TOKEN }}
        CHAT_ID: ${{ secrets.CHAT_ID }}
